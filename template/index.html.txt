  <script>
    // State Aplikasi
    let statusAdminAkses = false;
    let dataAdminUtama = []; 
    let dataAdminTampil = []; 
    let kolomSortAktif = -1;
    let arahSortAsc = true;
    
    let base64GambarKamera = '';
    let linkGambarRTServer = ''; 

    // Referensi Alamat
    const referensiAlamat = {
      'Gunungguruh': { rt: ['01', '02', '03', '04', '05', '06', '07', '08', '09'], rw: ['01', '02', '03', '04'] },
      'Bojongduren': { rt: ['10', '11', '12', '13', '14', '15', '16'], rw: ['05', '06', '07'] },
      'Limusnunggal': { rt: ['17', '18', '19', '20', '21', '22'], rw: ['08', '09', '10'] }
    };

    document.addEventListener("DOMContentLoaded", () => {
      if(localStorage.getItem('sesiAdminDesaCibentang') === 'aktif') {
        statusAdminAkses = true;
        document.getElementById('ikon-kunci').classList.replace('fa-lock', 'fa-unlock');
      }
    });

    // ===============================================
    // LOGIKA KAMERA & KOMPRESI PINTAR (500KB - 1MB)
    // ===============================================
    function prosesGambarKamera(event) {
      const file = event.target.files[0];
      if (!file) return;

      const placeholder = document.getElementById('upload-placeholder');
      const preview = document.getElementById('preview-gambar');
      const btnLanjut = document.getElementById('btn-lanjut-form');
      
      const infoKompresi = document.getElementById('info-kompresi');
      const ikonStatus = document.getElementById('status-ikon');
      const teksStatus = document.getElementById('status-teks');
      const spanUkuran = document.getElementById('ukuran-gambar');

      // Tampilkan status memuat
      placeholder.innerHTML = `<i class="fa-solid fa-circle-notch fa-spin text-4xl text-blue-500 mb-3"></i><p class="text-sm font-semibold text-blue-600">Menganalisis & Mengompresi...</p>`;
      infoKompresi.classList.add('hidden'); // Sembunyikan info lama jika ada

      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = function(e) {
        const img = new Image();
        img.src = e.target.result;
        img.onload = function() {
          
          // Logika Kompresi Pintar untuk mencapai 500KB - 1MB
          let resultBase64 = '';
          let finalKB = 0;
          let iterations = 0;
          const MAX_ITERATIONS = 5;
          
          // Parameter awal
          let currentWidth = img.width;
          let currentHeight = img.height;
          let currentQuality = 0.90; // Mulai dari kualitas tinggi

          // 1. Cek ukuran asli terlebih dahulu
          resultBase64 = e.target.result; // Gunakan aslinya dulu
          finalKB = Math.round((resultBase64.length * 3 / 4) / 1024);

          // 2. Jika terlalu besar (> 1MB), lakukan kompresi iteratif ke bawah
          if (finalKB > 1024) {
            console.log("Gambar terlalu besar, mulai kompresi ke bawah...");
            
            // Tentukan batas resolusi maksimal (misal 3000px agar tidak terlalu masif)
            const MAX_RES = 3000;
            if (currentWidth > MAX_RES || currentHeight > MAX_RES) {
                if (currentWidth > currentHeight) { currentHeight *= MAX_RES / currentWidth; currentWidth = MAX_RES; } 
                else { currentWidth *= MAX_RES / currentHeight; currentHeight = MAX_RES; }
            }

            for (iterations = 0; iterations < MAX_ITERATIONS; iterations++) {
              const canvas = document.createElement('canvas');
              canvas.width = currentWidth;
              canvas.height = currentHeight;
              const ctx = canvas.getContext('2d');
              ctx.drawImage(img, 0, 0, currentWidth, currentHeight);
              
              resultBase64 = canvas.toDataURL('image/jpeg', currentQuality);
              finalKB = Math.round((resultBase64.length * 3 / 4) / 1024);

              console.log(`Iterasi ${iterations+1} (Down): Res=${currentWidth}x${currentHeight}, Qual=${currentQuality}, Size=${finalKB}KB`);

              if (finalKB <= 1000) break; // Berhenti jika sudah di bawah 1MB

              // Turunkan parameter untuk iterasi berikutnya
              currentQuality -= 0.1; // Turunkan kualitas 10%
              currentWidth *= 0.9;  // Turunkan resolusi 10%
              currentHeight *= 0.9;
              
              if (currentQuality < 0.5) currentQuality = 0.5; // Batas minimal kualitas
            }
          }
          
          // 3. Jika terlalu kecil (< 500KB) setelah Down-sampling atau memang aslinya kecil, 
          // Coba Up-sampling resolusi atau Qual untuk detail lebih baik (hanya jika Down-sampling tidak dilakukan)
          if (finalKB < 500 && iterations === 0) {
              console.log("Gambar terlalu kecil, menggunakan kualitas asli maksimal.");
              // Jika aslinya kecil, kita tidak bisa membuatnya lebih besar tanpa buram,
              // jadi kita gunakan kualitas tertinggi yang mungkin dari canvas.
              const canvas = document.createElement('canvas');
              canvas.width = img.width; // Gunakan res asli
              canvas.height = img.height;
              const ctx = canvas.getContext('2d');
              ctx.drawImage(img, 0, 0, img.width, img.height);
              
              resultBase64 = canvas.toDataURL('image/jpeg', 1.0); // Kualitas maksimal 100%
              finalKB = Math.round((resultBase64.length * 3 / 4) / 1024);
              console.log(`Final (Max Quality): Size=${finalKB}KB`);
          }

          // 4. Update Global Variable
          base64GambarKamera = resultBase64;
          
          // 5. Update UI
          preview.src = base64GambarKamera;
          preview.classList.remove('hidden');
          placeholder.classList.add('hidden');
          spanUkuran.innerText = finalKB;

          // Mengatur Warna & Ikon Status Ukuran
          infoKompresi.classList.remove('hidden');
          if (finalKB >= 500 && finalKB <= 1024) {
            // Sempurna (Hijau)
            infoKompresi.className = "mt-4 text-center text-sm font-medium py-2 rounded-xl border border-green-100 bg-green-50 text-green-600";
            ikonStatus.innerHTML = '<i class="fa-solid fa-circle-check mr-1"></i>';
            teksStatus.innerText = "Ukuran Terbaik Berhasil Dicapai";
          } else if (finalKB > 1024) {
             // Masih terlalu besar (Kuning)
            infoKompresi.className = "mt-4 text-center text-sm font-medium py-2 rounded-xl border border-yellow-100 bg-yellow-50 text-yellow-700";
            ikonStatus.innerHTML = '<i class="fa-solid fa-triangle-exclamation mr-1"></i>';
            teksStatus.innerText = "Selesai (Sedikit di atas 1MB)";
          } else {
             // Terlalu kecil (Kuning)
            infoKompresi.className = "mt-4 text-center text-sm font-medium py-2 rounded-xl border border-yellow-100 bg-yellow-50 text-yellow-700";
            ikonStatus.innerHTML = '<i class="fa-solid fa-image mr-1"></i>';
            teksStatus.innerText = "Selesai (Menggunakan Kualitas Maksimal)";
          }

          // Aktifkan tombol
          btnLanjut.disabled = false;
          btnLanjut.classList.replace('bg-gray-300', 'bg-blue-600');
          btnLanjut.classList.replace('text-gray-500', 'text-white');
          btnLanjut.classList.replace('cursor-not-allowed', 'hover:bg-blue-700');
        }
      }
    }

    function switchView(view) {
      const pengantarView = document.getElementById('view-pengantar');

      const suksesView = document.getElementById('view-sukses');

      adminView.classList.add('hidden');

        pengantarView.classList.remove('hidden');
        btnPemohon.classList.add('active');
        btnAdmin.classList.remove('active');
      } 
      else if (view === 'pemohon') {
        pemohonView.classList.remove('hidden');
        btnPemohon.classList.add('active');
        btnAdmin.classList.remove('active');
      }
      else if (view === 'sukses') {
        suksesView.classList.remove('hidden');
        btnPemohon.classList.add('active');
        btnAdmin.classList.remove('active');
      }
      else if (view === 'admin') {
        adminView.classList.remove('hidden');
        btnPemohon.classList.remove('active');
        btnAdmin.classList.add('active');
        muatDataAdmin(); 
      }
    }

        cancelButtonText: 'Batal',

    function muatDataAdmin() {
      const tbody = document.getElementById('tabel-body');
      tbody.innerHTML = `<tr><td colspan="5" class="py-12 text-center"><i class="fa-solid fa-spinner fa-spin text-3xl mb-3 text-blue-500"></i><p class="text-gray-500">Memuat data...</p></td></tr>`;
      document.getElementById('input-pencarian').value = '';
      kolomSortAktif = -1; resetIkonSort();

      google.script.run
        .withSuccessHandler(function(data) {
          if (!data || data.length === 0) { dataAdminUtama = []; dataAdminTampil = []; renderTabel(); return; }
          dataAdminUtama = data; dataAdminTampil = [...dataAdminUtama]; renderTabel();
        })
        .withFailureHandler(function(error) {
           tbody.innerHTML = `<tr><td colspan="5" class="py-12 text-center text-red-500">Gagal memuat data: ${error.message}</td></tr>`;
        })
        .dapatkanDataAdmin();
    }

    function renderTabel() {

        return;
      }

      let htmlRows = '';
      dataAdminTampil.forEach(function(row, index) {
        const tanggalStr = row[0] || '-';
        const nama = row[1] || '-';
        const nik = row[2] || '-';
        const jenis = row[3] || '-';
        const urlPdf = row[4] || '#';
        const urlDocx = row[5] || ''; 
        const urlGbr = row[6] || ''; 

        let tombolAksi = `<div class="flex items-center justify-center space-x-2">`;
        
        if (urlGbr !== '') {
          tombolAksi += `<a href="${urlGbr}" target="_blank" class="inline-flex items-center justify-center bg-yellow-50 hover:bg-yellow-500 hover:text-white text-yellow-600 p-2.5 rounded-xl transition-all duration-200 shadow-sm border border-yellow-100" title="Lihat Foto RT"><i class="fa-solid fa-image text-lg"></i></a>`;
        }
        
        tombolAksi += `<a href="${urlPdf}" target="_blank" class="inline-flex items-center justify-center bg-red-50 hover:bg-red-500 hover:text-white text-red-600 p-2.5 rounded-xl transition-all duration-200 shadow-sm border border-red-100" title="Unduh PDF"><i class="fa-solid fa-file-pdf text-lg"></i></a>`;
        
        if (urlDocx !== '') {
          tombolAksi += `<a href="${urlDocx}" target="_blank" class="inline-flex items-center justify-center bg-blue-50 hover:bg-blue-600 hover:text-white text-blue-600 p-2.5 rounded-xl transition-all duration-200 shadow-sm border border-blue-100" title="Unduh Word"><i class="fa-solid fa-file-word text-lg"></i></a>`;

        `;
      });

    }

    function cariDataAdmin() {
      const keyword = document.getElementById('input-pencarian').value.toLowerCase();
      dataAdminTampil = dataAdminUtama.filter(row => {
        const digabung = (row[0] + " " + row[1] + " " + row[2] + " " + row[3]).toLowerCase();
        return digabung.includes(keyword);
      });
      if(kolomSortAktif !== -1) { jalankanSort(kolomSortAktif, arahSortAsc); } else { renderTabel(); }
    }
    function sortTabel(indeksKolom) {
      if (kolomSortAktif === indeksKolom) { arahSortAsc = !arahSortAsc; } else { kolomSortAktif = indeksKolom; arahSortAsc = true; }
      resetIkonSort();
      const iconElement = document.getElementById(`sort-icon-${indeksKolom}`);
      if (arahSortAsc) { iconElement.className = 'fa-solid fa-sort-up ml-1 text-blue-500 transition'; } 
      else { iconElement.className = 'fa-solid fa-sort-down ml-1 text-blue-500 transition'; }
      jalankanSort(indeksKolom, arahSortAsc);
    }
    function jalankanSort(indeksKolom, ascending) {
      dataAdminTampil.sort((a, b) => {
        let valA = a[indeksKolom] ? a[indeksKolom].toString().toLowerCase() : '';
        let valB = b[indeksKolom] ? b[indeksKolom].toString().toLowerCase() : '';
        if (valA < valB) return ascending ? -1 : 1;
        if (valA > valB) return ascending ? 1 : -1; return 0;
      });
      renderTabel();
    }
    function resetIkonSort() {
      for (let i = 0; i <= 3; i++) {
        const icon = document.getElementById(`sort-icon-${i}`);
        if(icon) icon.className = 'fa-solid fa-sort ml-1 text-gray-400 transition';
      }
    }
